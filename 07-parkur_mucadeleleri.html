// BÖLÜM 01

QTR-8A Sensör — Çalışma Mantığı

QTR-8A, satır içinde yan yana dizilmiş 8 adet kızılötesi (IR) alıcı-verici içeren bir reflektans sensör dizisidir. Her sensör altındaki yüzeyden geri dönen ışık miktarını ölçer ve analog gerilim değeri olarak Arduino'ya iletir.

⚡

Analog Çıkış

Her sensör 0–1023 arası bir değer üretir. Beyaz yüzey → düşük değer, siyah yüzey → yüksek değer.

📍

Konum Hesabı

readLineBlack() 0–5000 arası ağırlıklı ortalama döndürür. Ortada çizgi → ~2500.

🔧

Kalibrasyon

Sensörü hem siyah hem beyaz yüzey üzerinde gezdirerek min/max referans değerleri belirlenir.

🔌

Bağlantı

6 sensör A0–A5 pinlerine bağlanır. Kütüphane: QTRSensors.h

P ve PD Kontrol Formülleri

Çizginin merkezden sapmasına hata denir. Bu hatayı düzeltmek için iki yöntem kullanılır:

// Hata hesabı (merkez = 2500)
int hata = konum - 2500;

// P (Oransal) Kontrol
int duzeltme = Kp * hata;

// PD (Oransal + Türev) Kontrol
int duzeltme = (Kp * hata) + (Kd * (hata - oncekiHata));
oncekiHata = hata;

// Motor hızlarına uygulanması
int solHiz = temelHiz + duzeltme;
int sagHiz = temelHiz - duzeltme;
💡

PD'nin farkı nedir? P kontrol sadece anlık hataya tepki verir; virajlarda aşma (overshoot) yapar, robot zikzak çizer. PD kontrolde eklenen türev terimi (Kd × Δhata) hatanın ne kadar hızlı değiştiğini ölçer ve gereksiz salınımları bastırır. Sonuç: daha akıcı, daha kararlı bir çizgi takibi.

ParametreTipik DeğerRolü
Kp0.07 – 0.12Anlık hataya tepki katsayısı. Büyütülürse hız artar ama titreme de artar.
Kd0.15 – 0.25Türev (değişim hızı) katsayısı. Salınımları söndürür.
temelHiz110 – 150Düz çizgideki ilerleme hızı (0–255 arası PWM).
cizgiKayipEsigi~300Tüm sensörlerin toplamı bu değerin altına düşerse çizgi kayıp sayılır.
// BÖLÜM 02

Kalibrasyon

Kalibrasyon, sensörlerin gerçek zemindeki beyaz ve siyah rengi öğrenmesini sağlar. Robot koda yüklendikten sonra sensörler hem beyaz hem siyah zemin üzerinden geçirilir; bu süre zarfında kütüphane min/max değerleri kaydeder. Sonraki tüm okumalar bu referanslara göre normalize edilir.

⚠️

Önemli: Kalibrasyon sırasında robotu yavaşça hem siyah çizginin üzerinden hem de beyaz zeminin üzerinden gezdirin. 200 × 20ms = 4 saniye sürer. Kalibrasyon bittikten sonra Serial Monitor'de sensör değerlerini izleyin. Beyaz zeminde değerler düşük (~50–200), siyah üzerinde yüksek (~700–1000) olmalıdır. Sensör tamamen beyaz zemindeyken ve siyah zemindeyken toplam sensör değerlerini not edin. Ardından,

cizgiKayipEsigi = (beyazToplam + siyahToplam) / 2

formülüyle eşiği belirleyin.

Arduino — 00_kalibrasyon.ino 
// cizgiKayipEsigi = (beyazToplam + cizgiToplam) / 2;

#include <QTRSensors.h>

QTRSensors qtr;

uint16_t sensorDegerleri[6];

void setup() {
  Serial.begin(115200);

  qtr.setTypeAnalog();
  uint8_t pins[] = {A0, A1, A2, A3, A4, A5};
  qtr.setSensorPins(pins, 6);

  // Kalibrasyon
  for (int i = 0; i < 200; i++) {
    qtr.calibrate();
    delay(20);
  }

  Serial.println("Sensor degerleri:");
}

void loop() {
  qtr.readLineBlack(sensorDegerleri);

  int toplamDeger = 0;

  for (int i = 0; i < 6; i++) {
    Serial.print(sensorDegerleri[i]);
    Serial.print("\t");
    toplamDeger += sensorDegerleri[i];
  }

  Serial.print("Toplam: ");
  Serial.println(toplamDeger);

  delay(100);
}
// PARKUR 01

Temel Çizgi Takibi

Robot, başlangıçtan bitişe kadar tek bir eğri çizgi üzerinde ilerler. Asıl hedef gereksiz zikzak yapmadan akıcı ilerlemektir.

01

QTR-8A ile Temel Çizgi Takibi

Kalibrasyon + P / PD Kontrol

  • readLineBlack() ile anlık çizgi konumu okunur (0–5000).
  • hata = konum − 2500 hesaplanır. Negatif → çizgi solda, pozitif → çizgi sağda.
  • P: duzeltme = Kp × hata ile sol/sağ motor hızları ayarlanır.
  • PD: Ek olarak türev terimi (Kd × Δhata) eklenerek titreme bastırılır.
  • Hızlar constrain(0, 255) ile sınırlandırılır, motorlar sürülür.
01_temel_cizgi_takibi_p.ino — loop() 
// Temel Çizgi Takibi - P

#include <QTRSensors.h>

QTRSensors qtr;

const uint8_t ENA = 6, IN1 = 7, IN2 = 8;    // Sağ motor
const uint8_t IN3 = 9, IN4 = 10, ENB = 11;  // Sol motor

float Kp = 0.08;      // S parkur için 0.10

int temelHiz = 130;   // S parkur için 120

uint16_t sensorDegerleri[6];

void setup() {
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);

  qtr.setTypeAnalog();
  uint8_t pinler[] = { A0, A1, A2, A3, A4, A5 };
  qtr.setSensorPins(pinler, 6);

  for (int i = 0; i < 200; i++) {
    qtr.calibrate();
    delay(20);
  }
}

void loop() {
  int konum = qtr.readLineBlack(sensorDegerleri);

  // Çizgi soldaysa hata negatif, sağdaysa pozitif olur
  int hata = konum - 2500;

  int duzeltme = Kp * hata;

  int solHiz = temelHiz + duzeltme;
  int sagHiz = temelHiz - duzeltme;

  solHiz = constrain(solHiz, 0, 255);
  sagHiz = constrain(sagHiz, 0, 255);

  motorSur(solHiz, sagHiz);
}

void motorSur(int solHiz, int sagHiz) {
  analogWrite(ENA, abs(sagHiz));
  analogWrite(ENB, abs(solHiz));

  if (sagHiz < 0) {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, HIGH);
  } else {
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
  }

  if (solHiz < 0) {
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, HIGH);
  } else {
    digitalWrite(IN3, HIGH);
    digitalWrite(IN4, LOW);
  }
}

// PARKUR 02

Keskin Dönüş ve Çizgi Kaybı Toparlama

Robot köşeye geldiğinde çizgiyi tüm sensörlerde kaybedebilir. Bu durumda rastgele davranmak yerine en son çizgiyi hangi tarafta gördüğünü hatırlaması gerekir.

02

Keskin Dönüş ve Çizgi Kaybı Toparlama

sonYon değişkeni + dönüş tetikleyicileri

🧠

Yeni kavram — sonYon: Her döngüde, hata −200'den küçükse sonYon = -1 (sol), 200'den büyükse sonYon = +1 (sağ) olarak güncellenir. Çizgi tamamen kaybolduğunda (toplamDeger < eşik) robot durmuyor; son yöne dönme hareketi yapıyor.

  • Her döngüde toplamDegerOku() ile toplam sensör değeri hesaplanır.
  • Toplam < cizgiKayipEsigi ise çizgi yok: sonYon'a göre yerinde dön.
  • Sol uç sensörler (S0, S1) siyah görüyorsa → keskin sola dön: motorSur(-140, 140).
  • Sağ uç sensörler (S4, S5) siyah görüyorsa → keskin sağa dön: motorSur(140, -140).
  • Normal durumda P veya PD kontrolle devam et.
03_90_derece_donus_p.ino — loop() 
// 90 derece dönüş - P

#include <QTRSensors.h>

QTRSensors qtr;

const uint8_t ENA = 6, IN1 = 7, IN2 = 8;    // Sağ motor
const uint8_t IN3 = 9, IN4 = 10, ENB = 11;  // Sol motor

float Kp = 0.08;

int temelHiz = 130;
int aramaHizi = 120;
int donusHizi = 140;
int cizgiKayipEsigi = 500;

uint16_t sensorDegerleri[6];

int sonYon = 0;  // -1 sol, +1 sağ

void setup() {
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);

  qtr.setTypeAnalog();
  uint8_t pinler[] = { A0, A1, A2, A3, A4, A5 };
  qtr.setSensorPins(pinler, 6);

  for (int i = 0; i < 200; i++) {
    qtr.calibrate();
    delay(20);
  }
}

void loop() {
  int konum = qtr.readLineBlack(sensorDegerleri);
  int toplamDeger = toplamDegerOku();

  // Çizgi tamamen beyaz — son yöne dönmeye çalış
  if (toplamDeger < cizgiKayipEsigi) {
    if (sonYon == -1) {
      motorSur(-aramaHizi, aramaHizi);
    } else if (sonYon == 1) {
      motorSur(aramaHizi, -aramaHizi);
    } else {
      motorDur();
    }
    return;
  }

  int hata = konum - 2500;

  // Son yönü güncelle
  if (hata < -200) sonYon = -1;
  else if (hata > 200) sonYon = 1;

  // Keskin sola dönüş
  if (sensorDegerleri[0] > 700 && sensorDegerleri[1] > 700 && sensorDegerleri[4] < 400 && sensorDegerleri[5] < 400) {
    sonYon = -1;
    motorSur(-donusHizi, donusHizi);
    return;
  }

  // Keskin sağa dönüş
  if (sensorDegerleri[4] > 700 && sensorDegerleri[5] > 700 && sensorDegerleri[0] < 400 && sensorDegerleri[1] < 400) {
    sonYon = 1;
    motorSur(donusHizi, -donusHizi);
    return;
  }

  int duzeltme = Kp * hata;

  int solHiz = constrain(temelHiz + duzeltme, 0, 255);
  int sagHiz = constrain(temelHiz - duzeltme, 0, 255);

  motorSur(solHiz, sagHiz);
}

void motorSur(int solHiz, int sagHiz) {
  analogWrite(ENA, abs(sagHiz));
  analogWrite(ENB, abs(solHiz));

  if (sagHiz < 0) {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, HIGH);
  } else {
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
  }

  if (solHiz < 0) {
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, HIGH);
  } else {
    digitalWrite(IN3, HIGH);
    digitalWrite(IN4, LOW);
  }
}

void motorDur() {
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);

  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

int toplamDegerOku() {
  int toplamDeger = 0;
  for (int i = 0; i < 6; i++) {
    toplamDeger += sensorDegerleri[i];
  }
  return toplamDeger;
}

// PARKUR 03

Kavşakta Karar Verme — Sağa Dön

Robot bir kavşakla karşılaştığında belirli bir karar kuralı uygulamalıdır. Bu parkurda kural nettir: kavşak algılanınca sağ kola dön.

03

Kavşakta Karar Verme (Sağa Dön)

donusYapildiMi bayrağı + tetikleyici koşul

🔀

Kavşak tespiti: Sağ taraftaki 4 sensör (S2, S3, S4, S5) hepsi siyah görüyorsa kavşak vardır. Robot biraz ilerlendikten sonra (delay(120)) keskin sağa döner (delay(220)). donusYapildiMi bayrağı sayesinde bu manevra yalnızca bir kez yapılır.

  • toplamDegerOku() ile çizgi kontrol edilir; kayıpsa dur.
  • S2, S3, S4, S5 değerlerinin tümü 500'ü aşıyorsa ve henüz dönüş yapılmadıysa → kavşak algılandı.
  • Kısa ileri gidiş (120 ms) → keskin sağa dönüş (220 ms).
  • donusYapildiMi = true → bir daha bu bloğa girilmez.
  • Normal P / PD çizgi takibine devam et.

  if (!donusYapildiMi &&
      sensorDegerleri[2] > 500 &&
      sensorDegerleri[3] > 500 &&
      sensorDegerleri[4] > 500 &&
      sensorDegerleri[5] > 500) {

    motorSur(120, 120);
    delay(120);

    motorSur(140, -140);
    delay(220);

    donusYapildiMi = true;
    return;
  }

  int hata = konum - 2500;
  int duzeltme = (Kp * hata) + (Kd * (hata - oncekiHata));

  oncekiHata = hata;

  int solHiz = constrain(temelHiz + duzeltme, 0, 255);
  int sagHiz = constrain(temelHiz - duzeltme, 0, 255);

  motorSur(solHiz, sagHiz);
}

void motorSur(int solHiz, int sagHiz) {
  analogWrite(ENA, abs(sagHiz));
  analogWrite(ENB, abs(solHiz));

  if (sagHiz < 0) {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, HIGH);
  } else {
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
  }

  if (solHiz < 0) {
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, HIGH);
  } else {
    digitalWrite(IN3, HIGH);
    digitalWrite(IN4, LOW);
  }
}

void motorDur() {
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);

  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

int toplamDegerOku() {
  int toplamDeger = 0;
  for (int i = 0; i < 6; i++) {
    toplamDeger += sensorDegerleri[i];
  }
  return toplamDeger;
}

Engeli Aşma ve Devam Etme

Robot engel gördüğünde çarpmadan etrafından dolaşır ve çizgi takibine devam eder. Engel geçildikten sonra parkur bitiminde durur.

04

Engeli Aşma ve Devam Etme

QTR-8A + HY-SRF05 Ultrasonik + P Kontrol

📡

HY-SRF05 ile mesafe ölçümü: TRIG pinine 10 µs'lik sinyal gönderilir. ECHO pininden gelen pulseIn süresi / 58.2 ile cm'ye çevrilir. Mesafe 12 cm'nin altındaysa engeldenDolan() çağrılır. Bu fonksiyon delay tabanlı sabit bir manevra dizisi uygular: dur → sağa dön → ileri → sola dön → ileri → çizgi takibine dön.

  • Her döngü başında mesafeOlc() ile ultrasonik mesafe ölçülür.
  • Mesafe 0 < d < 12 cm ise → engeldenDolan() çağrılır (delay tabanlı manevra).
  • Engel yoksa QTR-8A ile çizgi takibine devam edilir.
  • Çizgi kaybolursa (toplamDeger < eşik) → motorDur() ile dur (parkur sonu).

// Engeli aş ve devam et - P

#include <QTRSensors.h>

QTRSensors qtr;

const uint8_t ENA = 6, IN1 = 7, IN2 = 8;    // Sağ motor
const uint8_t IN3 = 9, IN4 = 10, ENB = 11;  // Sol motor

// HY-SRF05 pinleri
const uint8_t trigPin = 4;
const uint8_t echoPin = 5;

// Engel algılama ayarı (cm)
int engelMesafeEsigi = 12;

// Çizgi takip ayarları
float Kp = 0.08;
int temelHiz = 130;
int cizgiKayipEsigi = 300;

uint16_t sensorDegerleri[6];

void setup() {
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);

  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  qtr.setTypeAnalog();
  uint8_t pinler[] = { A0, A1, A2, A3, A4, A5 };
  qtr.setSensorPins(pinler, 6);

  for (int i = 0; i < 200; i++) {
    qtr.calibrate();
    delay(20);
  }
}

void loop() {
  int mesafe = mesafeOlc();

  // Engel varsa etrafından dolaş
  if (mesafe > 0 && mesafe < engelMesafeEsigi) {
    engeldenDolan();
    return;
  }

  int konum = qtr.readLineBlack(sensorDegerleri);
  int toplamDeger = toplamDegerOku();

  // Çizgi kayıpsa dur
  if (toplamDeger < cizgiKayipEsigi) {
    motorDur();
    return;
  }

  int hata = konum - 2500;
  int duzeltme = Kp * hata;

  int solHiz = constrain(temelHiz + duzeltme, 0, 255);
  int sagHiz = constrain(temelHiz - duzeltme, 0, 255);

  motorSur(solHiz, sagHiz);
}

int mesafeOlc() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);

  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long sure = pulseIn(echoPin, HIGH, 25000);

  if (sure == 0) {
    return 0;
  }

  return sure / 58.2;
}

void engeldenDolan() {
  motorDur();
  delay(200);

  // Sağa dön
  motorSur(130, -130);
  delay(350);

  // İleri git
  motorSur(130, 130);
  delay(500);

  // Sola dön
  motorSur(-130, 130);
  delay(350);

  // Biraz daha ileri git
  motorSur(130, 130);
  delay(500);
}

void motorSur(int solHiz, int sagHiz) {
  analogWrite(ENA, abs(sagHiz));
  analogWrite(ENB, abs(solHiz));

  if (sagHiz < 0) {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, HIGH);
  } else {
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
  }

  if (solHiz < 0) {
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, HIGH);
  } else {
    digitalWrite(IN3, HIGH);
    digitalWrite(IN4, LOW);
  }
}

void motorDur() {
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);

  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

int toplamDegerOku() {
  int toplamDeger = 0;
  for (int i = 0; i < 6; i++) {
    toplamDeger += sensorDegerleri[i];
  }
  return toplamDeger;
}
